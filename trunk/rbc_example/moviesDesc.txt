%Prelimanary Grammar
% @targetType URI(a rdf type)
%
% @target attribute_name
%
% @attribute name
% [URI(s) separated by commas]
% [NOMINAL|ENUM|BINNED]=[values separated by commas]
% aggregator=[NONE|INDEPENDENT_VAL|COUNT|AVG|MIN|MAX]
%
%Remark: Number is always binned
%Remark:If BINNED Option the format is range (of integers) seperated by commas
% Example: 3,5,10
% Bin index:
%  (-infty, 3) = 0
%  [3, 5) = 1
%  [5, 10) = 2
%  [10, +infty) = 3
%
% Example: 5
% Bin index:
%  (-infty, 5) = 0
%  [5, +infty) = 1
%Remark:It is assumed that the AVG,MIN,MAX are applicable to NUMBER datatype only.
%TODO add prefix support

@targetType http://data.linkedmdb.org/resource/movie/film

@target openingreceipt

@attribute openingreceipt
http://data.linkedmdb.org/resource/movie/openingreceipt
BINNED=2000000
aggregator=NONE

@attribute genre
http://data.linkedmdb.org/resource/movie/genre
ENUM=??
aggregator=INDEPENDENT_VAL

@attribute runtime
http://data.linkedmdb.org/resource/movie/runtime
BINNED=80,100,120,140,160
aggregator=NONE

@attribute actorBirth
http://data.linkedmdb.org/resource/movie/actor,http://xmlns.com/foaf/0.1/page,http://rdf.freebase.com/ns/people.person.year_of_birth
BINNED=1940,1960
aggregator=AVG

@attribute actorGender
http://data.linkedmdb.org/resource/movie/actor,http://xmlns.com/foaf/0.1/page,http://rdf.freebase.com/ns/people.person.gender
ENUM=http://rdf.freebase.com/ns/en.male,http://rdf.freebase.com/ns/en.female
aggregator=INDEPENDENT_VAL

@attribute directorAward
http://data.linkedmdb.org/resource/movie/director,http://xmlns.com/foaf/0.1/page,http://rdf.freebase.com/ns/award.award_winner.awards_won
BINNED=2
aggregator=COUNT

@attribute producerAward
http://data.linkedmdb.org/resource/movie/producer,http://xmlns.com/foaf/0.1/page,http://rdf.freebase.com/ns/award.award_winner.awards_won
BINNED=2
aggregator=COUNT
