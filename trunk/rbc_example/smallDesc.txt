%Prelimanary Grammar
% @targetType URI(a rdf type)
%
% @target attribute_name
%
% @attribute name
% [URI(s) separated by commas]
% [NOMINAL|ENUM|BINNED]=[values separated by commas]
% aggregator=[NONE|INDEPENDENT_VAL|COUNT|AVG|MIN|MAX]
%
%Remark: Number is always binned
%Remark:If BINNED Option the format is range (of integers) seperated by commas
% Example: 3,5,10
% Bin index:
%  (-infty, 3) = 0
%  [3, 5) = 1
%  [5, 10) = 2
%  [10, +infty) = 3
%
% Example: 5
% Bin index:
%  (-infty, 5) = 0
%  [5, +infty) = 1
%Remark:It is assumed that the AVG,MIN,MAX are applicable to NUMBER datatype only.
%TODO add prefix support

@targetType :small/Instance

@target label

@attribute label
:small/label
NOMINAL=A,B,C
aggregator=NONE

@attribute ind
:small/prop1,:small/prop2
NOMINAL=i1, i2, i3
aggregator=INDEPENDENT_VAL

@attribute count
:small/prop1,:small/prop2
BINNED=2,4
aggregator=COUNT

@attribute avg
:small/prop1,:small/prop3
BINNED=15,25,35
aggregator=AVG

@attribute min
:small/prop1,:small/prop3
BINNED=15,25,35
aggregator=MIN

@attribute max
:small/prop1,:small/prop3
BINNED=15,25,35
aggregator=MAX
